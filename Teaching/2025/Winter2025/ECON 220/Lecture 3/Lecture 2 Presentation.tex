% Slide 1: Introduction
\begin{frame}{Introduction}
    \begin{itemize}

        \begin{itemize}
            \item Global impacts of technology.
            \item Local impacts on Northern British Columbia (NBC).
        \end{itemize}
    \end{itemize}

% Slide 2: Defining Technological Change
\begin{frame}{Defining Technological Change}
    \begin{itemize}
        \item \textbf{Technological Change}: The process of invention, innovation, and diffusion of technology.
        \item Types:
        \begin{itemize}
            \item \textbf{Incremental innovations}: Small improvements (e.g., better smartphone cameras).
            \item \textbf{Radical innovations}: Disruptive technologies (e.g., AI, blockchain).
        \end{itemize}
    \end{itemize}
\end{frame}

% Slide 3: Historical Perspective
\begin{frame}{Historical Perspective}
    \begin{itemize}
        \item Industrial Revolutions:
        \begin{itemize}
            \item 1st: Steam power and mechanization.
            \item 2nd: Electricity and mass production.
            \item 3rd: Computers and automation.
            \item 4th: AI, IoT, and advanced robotics.
        \end{itemize}
        \item Each revolution disrupted labor markets, trade, and production.
    \end{itemize}
\end{frame}

% Slide 5: Productivity Growth
\begin{frame}{Productivity Growth: Explanation}
    \begin{itemize}
        \item \textbf{Definition}: Productivity is the output per unit of input.
        \item Technology increases productivity by:
        \begin{itemize}
            \item Automating tasks.
            \item Enhancing accuracy and speed.
        \end{itemize}
    \end{itemize}
\end{frame}

% Slide 6: Example of Productivity Growth
\begin{frame}{Example: Productivity Growth}
    \begin{itemize}
        \item Example: Agricultural Technology
        \begin{itemize}
            \item Tractors, GPS systems, and automated harvesters improve efficiency.
            \item Result: Reduced labor costs and increased crop yields.
        \end{itemize}
        \item Global Impact: More food supply and lower prices.
    \end{itemize}
\end{frame}

% Slide 9: Labor Market Transformations
\begin{frame}{Labor Market Transformations}
    \begin{itemize}
        \item Automation replaces routine jobs.
        \item Demand grows for tech-related roles (e.g., data analysts, software developers).
        \item Skills gap:
        \begin{itemize}
            \item Many workers lack training for high-skill jobs.
        \end{itemize}
    \end{itemize}
\end{frame}

% Slide 10: Labor Market Example
\begin{frame}{Labor Market Example: AI in Finance}
    \begin{itemize}
        \item AI automates tasks like fraud detection and credit risk assessment.
        \item Result: Increased efficiency but fewer entry-level analyst roles.
        \item New jobs: AI model developers and cybersecurity experts.
    \end{itemize}
\end{frame}

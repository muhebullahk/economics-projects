\begin{enumerate}


\item Consider the system:
\[
\begin{cases}
x^2 + y + z = \alpha \\
xy - z = \beta \\
y^2 + z^2 = 1
\end{cases}
\]
\begin{enumerate}[label=(\alph*)]
\item Write the Jacobian matrix of the system with respect to $(x,y,z)$.
\item State the condition under which the Implicit Function Theorem guarantees the existence of differentiable functions $(x(\alpha,\beta), y(\alpha,\beta), z(\alpha,\beta))$.
\item Derive an expression for $\dfrac{\partial (x,y,z)}{\partial \alpha}$ using Cramer’s rule.
\end{enumerate}

\item
Suppose the transformation is defined by:
\[
u = x^2 + y^2 + z^2, \quad v = xy + yz + zx, \quad w = xyz.
\]
\begin{enumerate}[label=(\alph*)]
\item Construct the Jacobian $\dfrac{\partial(u,v,w)}{\partial(x,y,z)}$.
\item Show that the determinant vanishes when at least two of $(x,y,z)$ are equal.
\item Explain the implication for whether the transformation is locally invertible.
\end{enumerate}
\newpage
\item
Let the system be:
\[
\begin{cases}
F_1(x,y,\theta) = x^2 + y^2 - \theta = 0 \\
F_2(x,y,\theta) = xy - 1 = 0
\end{cases}
\]
\begin{enumerate}[label=(\alph*)]
\item Verify that the Jacobian determinant with respect to $(x,y)$ is nonzero at $(x,y)=(\sqrt{2}, 1/\sqrt{2}), \theta=3$.
\item Use the Implicit Function Theorem to argue for local differentiability of $(x(\theta),y(\theta))$.
\item Compute $\dfrac{dx}{d\theta}, \dfrac{dy}{d\theta}$ at this point.
\end{enumerate}
\end{enumerate}

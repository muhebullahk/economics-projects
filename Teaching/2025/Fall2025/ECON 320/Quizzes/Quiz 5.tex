\noindent\textbf{{Multiple Choice Questions (1 mark each)}}

\begin{enumerate}

\item The \textbf{Hessian matrix} of a function $f(x,y)$ is:
\begin{enumerate}
\item A matrix of first partial derivatives
\item A matrix of second partial derivatives
\item A matrix of total differentials
\item The Jacobian of $f$
\end{enumerate}

\vspace{0.5em}

\item If $D = f_{xx}f_{yy} - (f_{xy})^2 > 0$ and $f_{xx} < 0$, the point $(x^*, y^*)$ is:
\begin{enumerate}
\item A saddle point
\item A local maximum
\item A local minimum
\item An inflection point
\end{enumerate}

\vspace{0.5em}

\item For $f(x, y) = x^2 - 2xy + 3y^2$, the determinant of the Hessian $D$ is:
\begin{enumerate}
\item $D = -8$
\item $D = 0$
\item $D = 2$
\item $D = 8$
\end{enumerate}

\vspace{0.5em}

\item The second-order condition for a local maximum of $f(x,y)$ is satisfied when:
\begin{enumerate}
\item $D > 0$ and $f_{xx} > 0$
\item $D < 0$
\item $D > 0$ and $f_{xx} < 0$
\item $f_{xy} = 0$
\end{enumerate}

\vspace{0.5em}

\item Suppose $f(x,y) = 4x^2 + 9y^2 - 12xy$. Which of the following statements is \textbf{true}?
\begin{enumerate}
\item The function has a maximum at $(0,0)$
\item The function has a minimum at $(0,0)$
\item The function has a saddle point at $(0,0)$
\item The determinant of the Hessian is zero at $(0,0)$
\end{enumerate}

\end{enumerate}

\vspace{1em}




\noindent \textbf{Problem 1 (6 marks)}\\
Let $f(x,y) = 3x^2 + xy + y^2 - 6x - 9y$.\\[4pt]
(a) Find all stationary points.\\
(b) Determine their nature (maximum, minimum, or saddle) using the Hessian determinant test.

\vspace{1em}

 \noindent \textbf{Problem 2 (9 marks)}\\
A firm’s production function is $Q = f(K,L) = 100K^{0.5}L^{0.5}$. The cost of capital is $r = 4$ and the wage rate is $w = 1$. The firm wants to produce 100 units of output.\\[4pt]
(a) Formulate the Lagrangian and find the first-order conditions.\\
(b) Solve for the cost-minimizing values of $K$ and $L$.\\
(c) Compute the minimum cost.\\
(Hint: use the property that at optimum, $\frac{MPL}{MPK} = \frac{w}{r}$.)

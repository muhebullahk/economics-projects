\section*{Multiple Choice Questions (20 points)}
Each question is worth 2 points.\\

\begin{enumerate}

    \item Suppose demand is \(Q = \sqrt{P+10} - I\). The derivative \(\partial Q/\partial I\) equals:
    \begin{enumerate}
        \item \(\frac{1}{2}(P+10)^{-1/2}\)
        \item \(-1\)
        \item \(0\)
        \item \(-\frac{1}{2}(P+10)^{-1/2}\)
    \end{enumerate}
\vspace{0.7cm}
    \item For \(z = x^2 y^3\), the total differential \(dz\) is:
    \begin{enumerate}
        \item \(2xy^3 dx + 3x^2 y^2 dy\)
        \item \(2xy dx + 3x^2 y dy\)
        \item \(2xy^3 + 3x^2 y^2\)
        \item \(dx + dy\)
    \end{enumerate}
\vspace{0.7cm}
    \item Let \(f(x)=4x - x^3\). At \(x=\sqrt{4/3}\), the function has:
    \begin{enumerate}
        \item A local maximum
        \item A local minimum
        \item An inflection point
        \item Cannot be determined
    \end{enumerate}
\vspace{0.7cm}
    \item For \(Y = A K^{0.5} L^{0.5}\), the elasticity of \(Y\) with respect to \(K\) is:
    \begin{enumerate}
        \item \(0.5\)
        \item \(1\)
        \item \(\frac{L}{K}\)
        \item \(0\)
    \end{enumerate}

    \item For \(f(x,y)=3x^2 + 4xy + y^2\), the Hessian implies:
    \begin{enumerate}
        \item Saddle point
        \item Local minimum
        \item Local maximum
        \item Cannot classify
    \end{enumerate}
\vspace{0.75cm}
    \item In \(\max U(x,y)=xy\) subject to \(x+y=10\), the Lagrange multiplier represents:
    \begin{enumerate}
        \item Marginal utility of \(x\)
        \item Marginal utility of \(y\)
        \item Marginal utility of relaxing the constraint
        \item Share of income allocated to \(x\)
    \end{enumerate}
\vspace{0.75cm}
    \item In \(\max_x f(x,a)=ax - x^2\), the envelope theorem gives:
    \begin{enumerate}
        \item \(x\)
        \item \(ax\)
        \item \(2ax\)
        \item \(x^2\)
    \end{enumerate}
\vspace{0.75cm}
    \item Consumer surplus for demand \(P(Q)=20 - Q\) at price \(P=10\) is:
    \begin{enumerate}
        \item 25
        \item 50
        \item 100
        \item 150
    \end{enumerate}
\vspace{0.75cm}
    \item The solution to \(\frac{dy}{dt}=4y\) with \(y(0)=3\) is:
    \begin{enumerate}
        \item \(y=3+4t\)
        \item \(y=3 e^{4t}\)
        \item \(y=12t\)
        \item \(y=4 e^{3t}\)
    \end{enumerate}
\vspace{0.75cm}
    \item If \(Y(t)=K^{0.3} L^{0.7} e^{0.02t}\), then:
    \begin{enumerate}
        \item \(\frac{dY}{dt} = 0.02Y\)
        \item \(\frac{dY}{dt} = 0.3Y\)
        \item \(\frac{dY}{dt} = 0.7Y\)
        \item \(\frac{dY}{dt} = 0\)
    \end{enumerate}

\end{enumerate}

\section*{Problems (80 points total)}

\subsection*{Problem 1 (16 points)}
A firm produces output  
\[
Q = x^{1/2} y^{1/3}.
\]

\begin{enumerate}
    \item Compute the total differential \(dQ\). \hfill (5 pts)
    \item If the firm controls only \(x\), compute \(dQ/dx\). \hfill (3 pts)
    \item Profit is \(\pi(x)=40Q - x - y - 5\). Holding \(y\) fixed, find the optimal \(x\). \hfill (5 pts)
    \item Determine whether the optimum is a maximum or minimum. \hfill (3 pts)
\end{enumerate}

\subsection*{Problem 2 (16 points)}
A technology firm's revenue evolves as  
\[
R(t)=500 e^{0.03 t}(1+t^{1/2}).
\]

\begin{enumerate}
    \item Compute \(\frac{dR}{dt}\). \hfill (5 pts)
    \item Compute the growth rate \(g(t)=\frac{1}{R}\frac{dR}{dt}\). \hfill (4 pts)
    \item Compute the elasticity of revenue with respect to time. \hfill (4 pts)
    \item Determine whether revenue growth accelerates or decelerates over time. \hfill (3 pts)
\end{enumerate}

\subsection*{Problem 3 (16 points)}
Consider  
\[
f(x,y)=20x - 4x^2 + 10y - 3y^2 - xy.
\]

\begin{enumerate}
    \item Compute the first-order partial derivatives. \hfill (4 pts)
    \item Solve the first-order conditions. \hfill (4 pts)
    \item Compute the Hessian matrix and its determinant at the critical point. \hfill (4 pts)
    \item Classify the critical point. \hfill (4 pts)
\end{enumerate}

\subsection*{Problem 4 (16 points)}
A consumer maximizes  
\[
U(x,y)=x^{1/2}y^{1/2}
\]
subject to
\[
p_x x + p_y y = I.
\]

\begin{enumerate}
    \item Set up the Lagrangian and derive the first-order conditions. \hfill (5 pts)
    \item Solve for \(x^*\) and \(y^*\). \hfill (5 pts)
    \item Use the envelope theorem to compute \(\frac{\partial V}{\partial I}\). \hfill (3 pts)
    \item Interpret the result. \hfill (3 pts)
\end{enumerate}

\subsection*{Problem 5 (16 points)}
Capital accumulation satisfies
\[
\frac{dK}{dt} = I(t) - \delta K,
\]
where
\[
I(t)=50 e^{0.01 t}, \qquad \delta = 0.05.
\]

\begin{enumerate}
    \item Solve the differential equation for \(K(t)\). \hfill (6 pts)
    \item Find the particular solution when \(K(0)=100\). \hfill (4 pts)
    \item Compute \(\displaystyle \int_0^{20} I(t)\, dt\). \hfill (3 pts)
    \item Interpret the long-run behavior of \(K(t)\). \hfill (3 pts)
\end{enumerate}

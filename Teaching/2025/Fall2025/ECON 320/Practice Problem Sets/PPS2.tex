\begin{enumerate}


\item Consider the system:
\[
\begin{cases}
x + y = \alpha \\
x - y = \beta
\end{cases}
\]
\begin{enumerate}[label=(\alph*)]
\item Solve for $x(\alpha,\beta)$ and $y(\alpha,\beta)$.
\item Compute $\dfrac{\partial x}{\partial \alpha}, \dfrac{\partial x}{\partial \beta}, \dfrac{\partial y}{\partial \alpha}, \dfrac{\partial y}{\partial \beta}$.
\item Interpret the effect of $\alpha$ and $\beta$ on the equilibrium values of $x$ and $y$.
\end{enumerate}


\item Let the implicit function be defined by:
\[
F(x,y,p) = x^2 + xy + y^2 - p = 0
\]
\begin{enumerate}[label=(\alph*)]
\item Use total differentiation to compute $\dfrac{\mathrm{d}x}{\mathrm{d}p}$ and $\dfrac{\mathrm{d}y}{\mathrm{d}p}$.
\item At the point $(x,y)=(1,1), p=3$, determine the signs of the comparative static effects.
\end{enumerate}


\item Consider the nonlinear system:
\[
\begin{cases}
x^2 + y = p \\
x + y^2 = q
\end{cases}
\]
\begin{enumerate}[label=(\alph*)]
\item Write the Jacobian matrix $J(x,y)$.
\item Compute $\det(J(x,y))$.
\item State the condition under which the system has a locally unique solution $(x(p,q),y(p,q))$.
\end{enumerate}


\item The system is:
\[
\begin{cases}
F_1(x,y,\theta) = x^2 + y - \theta = 0 \\
F_2(x,y,\theta) = xy - 1 = 0
\end{cases}
\]
\begin{enumerate}[label=(\alph*)]
\item Compute the Jacobian matrix with respect to $(x,y)$.
\item Show that the determinant is nonzero at $(x,y)=(1,1), \theta=2$.
\item Apply the Implicit Function Theorem to argue about existence and differentiability of $(x(\theta), y(\theta))$.
\item Compute $\dfrac{dx}{d\theta}, \dfrac{dy}{d\theta}$ at $\theta=2$.
\end{enumerate}


\item Suppose:
\[
\begin{cases}
x + y + z = \theta \\
x^2 + y^2 = 1 \\
z^2 = x
\end{cases}
\]
\begin{enumerate}[label=(\alph*)]
\item Write the Jacobian matrix of the system with respect to $(x,y,z)$.
\item Use Cramer’s Rule to find $\dfrac{\partial(x,y,z)}{\partial \theta}$.
\item Discuss the effect of a marginal change in $\theta$ on the three variables.
\end{enumerate}


\item Let the transformation be:
\[
u = x^2 + y^2, \quad v = xy
\]
\begin{enumerate}[label=(\alph*)]
\item Compute the Jacobian $\dfrac{\partial(u,v)}{\partial(x,y)}$.
\item If possible, compute the inverse Jacobian $\dfrac{\partial(x,y)}{\partial(u,v)}$.
\item Explain how this transformation could simplify optimization problems with quadratic forms.
\end{enumerate}
\end{enumerate}

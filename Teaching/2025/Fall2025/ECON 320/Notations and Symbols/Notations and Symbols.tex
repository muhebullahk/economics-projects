\begin{table}[ht]

\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
\textbf{Symbol} & \textbf{Definition} & \textbf{Example} \\
\midrule
$\mathbb{N}$ & Set of natural numbers & $\mathbb{N} = \{1,2,3,\dots\}$ \\
$\mathbb{N}_0$ & Natural numbers including zero & $\mathbb{N}_0 = \{0,1,2,3,\dots\}$ \\
$\mathbb{Z}$ & Set of integers & $\mathbb{Z} = \{\dots,-1,0,1,\dots\}$ \\
$\mathbb{Q}$ & Set of rational numbers & $\frac{1}{2} \in \mathbb{Q}$ \\
$\mathbb{R}$ & Set of real numbers & $\sqrt{2} \in \mathbb{R}$ \\
$\mathbb{C}$ & Set of complex numbers & $1+i \in \mathbb{C}$ \\
$\in$ & Element of a set & $x \in \mathbb{R}$ \\
$\notin$ & Not element of a set & $\pi \notin \mathbb{Q}$ \\
$\subset, \subseteq$ & Subset, subset or equal & $A \subseteq B$ \\
$\supset, \supseteq$ & Superset, superset or equal & $B \supseteq A$ \\
$\cap$ & Intersection of sets & $A \cap B$ \\
$\cup$ & Union of sets & $A \cup B$ \\
$\emptyset$ & Empty set & $A \cap \emptyset = \emptyset$ \\
$\forall$ & For all (universal quantifier) & $\forall x \in \mathbb{R}, x^2 \ge 0$ \\
$\exists$ & There exists (existential quantifier) & $\exists x \in \mathbb{R}: x^2 = 4$ \\
$\Rightarrow$ & Implies & $x > 1 \Rightarrow x^2 > 1$ \\
$\Leftrightarrow$ & If and only if (iff) & $x^2=1 \Leftrightarrow x=\pm 1$ \\
$|A|$ & Cardinality of a set & If $A=\{1,2,3\}$, then $|A|=3$ \\
$\mathbb{P}(A)$ & Probability of event $A$ & $\mathbb{P}(A)=0.3$ \\
$\mathbb{E}[X]$ & Expectation of random variable $X$ & $\mathbb{E}[X]=\sum x_i p_i$ \\
$\mathrm{Var}(X)$ & Variance of $X$ & $\mathrm{Var}(X)=\mathbb{E}[(X-\mathbb{E}[X])^2]$ \\
$\sigma$ & Standard deviation & $\sigma = \sqrt{\mathrm{Var}(X)}$ \\
$\Delta x$ & Change or difference in $x$ & $\Delta x = x_2-x_1$ \\
$\partial f/\partial x$ & Partial derivative of $f$ wrt $x$ & $\partial(xy)/\partial x = y$ \\
$\nabla f$ & Gradient of $f$ & $\nabla f=(\partial f/\partial x,\partial f/\partial y)$ \\
$\int f(x)\,dx$ & Integral of $f(x)$ & $\int x\,dx = \tfrac{x^2}{2}+C$ \\
$\sum_{i=1}^n a_i$ & Summation notation & $\sum_{i=1}^3 i = 6$ \\
$\prod_{i=1}^n a_i$ & Product notation & $\prod_{i=1}^3 i = 6$ \\
$\pi$ & Archimedes' constant & $\pi \approx 3.1416$ \\
$e$ & Euler’s number (base of $\ln$) & $e \approx 2.718$ \\

\bottomrule

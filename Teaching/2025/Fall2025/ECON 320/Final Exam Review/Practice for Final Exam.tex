% ---------- PROBLEM 1 ----------
\section*{Problem 1 (Functions and Graphs)}

Consider the function
\[
f(x)=\frac{3x^2 - 4x + 5}{x+2}.
\]

\begin{enumerate}
    \item Determine the domain of \(f(x)\).
    \item Compute the horizontal and vertical asymptotes, if any.
    \item Sketch the basic shape of the function and identify any intercepts.
\end{enumerate}




% ---------- PROBLEM 2 ----------
\section*{Problem 2 (Linear Models and Elasticity)}

A market demand function is given by
\[
Q_d = 120 - 2P,
\qquad
Q_s = -20 + 3P.
\]

\begin{enumerate}
    \item Find the equilibrium price and quantity.
    \item Compute the price elasticity of demand at the equilibrium.
    \item Determine the price elasticity of supply at the equilibrium.
\end{enumerate}

% ---------- PROBLEM 4 ----------
\section*{Problem 4 (Single-Variable Optimization)}

A firm has the cost function
\[
C(Q)=20Q + 5Q^2,
\]
and the inverse demand function
\[
P(Q)=100 - Q.
\]

\begin{enumerate}
    \item Write the profit function \(\pi(Q)\).
    \item Find the first-order condition for profit maximization.
    \item Determine the critical point(s).
    \item State whether each critical point is a maximum or minimum (you do not need to justify).
\end{enumerate}



% ---------- PROBLEM 5 ----------
\section*{Problem 5 (Exponential and Logarithmic Functions)}

Suppose a country's GDP evolves according to
\[
Y(t)=500 e^{0.04t}.
\]

\begin{enumerate}
    \item Compute the instantaneous growth rate of \(Y(t)\).
    \item Find \(Y(20)\).
    \item Calculate the elasticity of \(Y\) with respect to \(t\).
\end{enumerate}



% ---------- PROBLEM 6 ----------
\section*{Problem 6 (Multivariable Optimization)}

Let
\[
f(x,y)=20x + 10y - x^2 - y^2 - 2xy.
\]

\begin{enumerate}
    \item Compute the first-order partial derivatives \(f_x\) and \(f_y\).
    \item Find all stationary points.
    \item Compute the Hessian matrix.
    \item Evaluate the determinant of the Hessian at the stationary point(s).
    \item Classify each stationary point as a maximum, minimum, or saddle point.
\end{enumerate}



% ---------- PROBLEM 7 ----------
\section*{Problem 7 (Constrained Optimization)}

A consumer maximizes the utility function
\[
U(x,y)=x^{1/3} y^{2/3}
\]
subject to the budget constraint
\[
4x + 2y = 120.
\]

\begin{enumerate}
    \item Form the Lagrangian for this problem.
    \item Derive the first-order conditions.
    \item Solve for the optimal bundle \((x^\ast, y^\ast)\).
    \item Write the bordered Hessian matrix for this problem (no need to evaluate it).
\end{enumerate}

% ---------- PROBLEM 8 ----------
\section*{Problem 8 (Envelope Theorem)}

Consider the maximization problem
\[
\max_{x} \; f(x,b)=bx - x^2,
\]
where \(b\) is a parameter, and there are no constraints on \(x\).

\begin{enumerate}
    \item Find the first-order condition for interior solutions.
    \item Derive the optimal \(x^\ast(b)\).
    \item Compute the value function \(v(b) = f(x^\ast(b), b)\).
    \item Use the envelope theorem to compute \(v'(b)\) without differentiating \(x^\ast(b)\).
\end{enumerate}

% ---------- PROBLEM 10 ----------
\section*{Problem 10 (Economic Dynamics)}

Consider the differential equation
\[
\frac{dy}{dt} + 3y = 12.
\]

\begin{enumerate}
    \item Solve the differential equation for the general solution \(y(t)\).
    \item Suppose \(y(0)=0\). Find the particular solution.
    \item Interpret the long-run behavior of \(y(t)\).
\end{enumerate}
